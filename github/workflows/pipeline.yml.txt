name: Daily Agro Data Pipeline

on:
  schedule:
    # Lance le script tous les jours à 08h00 (Heure UTC)
    - cron: '0 8 * * *'
  workflow_dispatch: # Permet de lancer le robot manuellement via un bouton sur GitHub

jobs:
  run-pipeline:
    runs-on: ubuntu-latest

    steps:
    - name: 📥 Récupération du code
      uses: actions/checkout@v3

    - name: 🐍 Installation de Python
      uses: actions/setup-python@v4
      with:
        python-version: '3.10'

    - name: 📦 Installation des librairies
      run: |
        pip install -r requirements.txt

    - name: 🚀 Exécution du script de données
      env:
        # On injecte la clé API cachée dans le coffre-fort
        NEWS_API_KEY: ${{ secrets.NEWS_API_KEY }}
      run: |
        python data_pipeline.py

    - name: 💾 Sauvegarde de la base de données
      run: |
        git config --local user.email "action@github.com"
        git config --local user.name "GitHub Action"
        # On force l'ajout de la base de données mise à jour (même si elle est dans le .gitignore)
        git add -f agri_data.db
        git commit -m "Mise à jour automatique des données boursières et NLP" || echo "Aucune nouvelle donnée"
        git push
